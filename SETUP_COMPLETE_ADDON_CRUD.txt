POST Operations (requires admin/manager role):
  POST body: { operation: 'update_addon', id, category, name, type, tiers/price }
  POST body: { operation: 'delete_addon', category, id }

  POST body: { operation: 'update_category', category, data: [...] }
  POST body: { operation: 'update_all_categories', data: {...categories} }
  POST body: { operation: 'reset_addons' }

Response Format:
  { success: true, message: '...', data: {...} }
  { success: false, error: '...', message: '...' }
